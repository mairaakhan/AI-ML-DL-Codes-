# MATHEMATICAL CONVERSIONS
# ==============================

# Celsius to Fahrenheit conversion
celsius = 37.5
fahrenheit = (celsius * 1.8) + 32
print(f"{celsius} Celsius is equal to {fahrenheit} Fahrenheit")

# Triangle area using Heron's formula
a = 5
b = 6
c = 7
s = (a + b + c) / 2
area = (s*(s-a)*(s-b)*(s-c)) ** 0.5
print(f"The area of triangle is {area}")

# Quadratic equation solution
import cmath
a = 1
b = 5
c = 6
d = (b**2) - (4*a*c)
sol1 = (-b-cmath.sqrt(d))/(2*a)
sol2 = (-b+cmath.sqrt(d))/(2*a)
print(sol1, sol2)

    print(num)

# Range loop

    print(i)

# String iteration
text = "Python"
for ch in text:
    print(ch)

# Nested loop
for i in range(2):
    for j in range(2):
        print(i, j)

# Break example
for ch in "abcdef":
    if ch == "d":
        break
    print(ch)

# Continue example
for ch in "abcdef":
    if ch == "c":
        continue
    print(ch)

# Pass statement

# Continue in while loop
i = 0
a = "navetstc"
while i < len(a):
    if a[i] == 'e' or a[i] == 's':
        i += 1
        continue
    print(a[i])
    i += 1

# ==============================
# SEARCHING
# ==============================

a = [1, 3, 5, 7, 9, 11]
val = 7
for i in a:
    if i == val:
        print("Found!")
        break
else:
    print("Not found")

# Countdown loop
cnt = 5
while True:
    print(cnt)
    cnt -= 1
    if cnt == 0:
        print("Countdown finished!")
        break

# Matrix search
matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
]
val = 5
found = False

for r in matrix:
    for n in r:
        if n == val:
            print(f"{val} found!")
            found = True
            break
    if found:
        break
