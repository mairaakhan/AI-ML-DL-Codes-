#It finds the most frequently used words in your text.
#FreqDist = used to count how often words appear
#.most_common(10) → shows top repeated words

from nltk.probability import FreqDist

#This code:
#Removes punctuation
#Keeps only alphabet words
#Converts everything to lowercase
#Counts final clean words

words_no_punc = []

for w in words:
    if w.isalpha():
        words_no_punc.append(w.lower())

# Create a frequency distribution (counts how often each word appears)

# Show the 10 most common words and their counts

#WordCloud → creates word cloud
#NumPy → converts image into array
#Pillow (PIL) → reads image file
#Matplotlib → displays output

# Import NumPy for working with arrays
import numpy as np

# Import Image module to load image file
from PIL import Image

# Import WordCloud tool from NLTK ecosystem
from wordcloud import WordCloud

# Convert image into a NumPy array (this will act as a shape mask)
char_mask = np.array(Image.open('circle.png'))

# Create word cloud:
# - background_color = 'Black' sets background color
# - mask = char_mask makes word cloud take shape of the image
wordcloud = WordCloud(background_color='Black', mask=char_mask).generate(text)

# Create figure size (width, height)
plt.figure(figsize=(8, 8))

# Display the word cloud image
plt.imshow(wordcloud)

# Hide axis lines (x and y axis)
plt.axis('off')

# Show the final output
plt.show()

#Stemming converts words to their root form.
#It removes endings like:
#-ing
#-ed
#-es

So words become their base form

from nltk.stem import PorterStemmer

# Create stemmer object
porter = PorterStemmer()

word_list = ['Study', 'Studying', 'Studies', 'Studied']

# Apply stemming
for w in word_list:
    print(porter.stem(w))
